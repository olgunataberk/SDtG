GAME:
	KARMA_DECLARATION_SECTION SCRRIPT_SECTION

KARMA_DECLARATION_SECTION:
	karma_start_token newline_token ( IDENTIFIER_LIST ) karma_end_token

IDENTIFIER_LIST:
	identifier , IDENTIFIER_LIST
	| identifier

SCRIPT_SECTION:
	script_start_token TEXT_LINE_LIST script_end_token
	
TEXT_LINE_LIST:
	TEXT_LINE newline_token TEXT_LINE_LIST
	| TEXT_LINE

TEXT_LINE:
	( KARMA_EXPRESSION ) identifier : LINE
	
LINE:
	CHOICE_PROMPT
	| string_token
		
CHOICE_PROMPT:
	choice_start_token string_token CHOICE_LIST choice_end_token
	
CHOICE_LIST:
	CHOICE_EXPRESSION , CHOICE_LIST
	| CHOICE_EXPRESSION
	
CHOICE_EXPRESSION:
	string_token : KARMA_OPERATION_LIST
	| string_token
	
KARMA_OPERATION_LIST:
	KARMA_OPERATION KARMA_OPERATION_LIST
	| KARMA_OPERATION
	
KARMA_OPERATION:
	( identifier operator_token integer_token )	
	
KARMA_EXPRESSION:
	KARMA_TERM 
	| KARMA_EXPRESSION or_token KARMA_TERM
	
KARMA_TERM: 
	KARMA_Factor 
	| KARMA_Term and_token KARMA_FACTOR
	
KARMA_FACTOR: 
	KARMA_PRIMARY
	| not_token KARMA_PRIMARY
	
KARMA_PRIMARY: 
	KARMA_PREDICATE 
	| ( KARMA_EXPRESSION )
	
KARMA_PREDICATE:
	identifier relational_operator_token integer_token
